type TestData struct {
	Basic_Types struct {
		ParamString	string
		ParamBool	bool
		ParamInt	int
		ParamFloat	float64
	}	`json:"Basic-Types"`

	Array_Types struct {
		ArrayInt	[]int
		ArrayBool	[]bool
		ArrayFloat	[]float64
		ArrayString	[]string
		ArrayMixed	[]string
		ArrayObjects	[]ArrayObjectsItem
	}	`json:"Array-Types"`
}

type ArrayObjectsItem struct {
	Foo	string	`json:"foo"`
	Bar	int	`json:"bar"`
}

func initTestData() TestData {
	var s TestData

	s.Basic_Types.ParamString = "ParamValue \n \"test\""
	s.Basic_Types.ParamBool = true
	s.Basic_Types.ParamInt = 12
	s.Basic_Types.ParamFloat = 1.234

	s.Array_Types.ArrayInt = []int{1, 2, 3}
	s.Array_Types.ArrayBool = []bool{true, false, true}
	s.Array_Types.ArrayFloat = []float64{1.0, 2.0, 3.0}
	s.Array_Types.ArrayString = []string{"a", "b", "c"}
	s.Array_Types.ArrayMixed = []string{"a", "1", "2.0"}
	s.Array_Types.ArrayObjects = []ArrayObjectsItem{}

	return s
}
